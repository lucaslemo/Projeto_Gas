Vendas de hoje
SELECT *
FROM vendas
WHERE vendas.data_venda = CURDATE()

Produtos em estoque
SELECT *
FROM produtos_estoque
WHERE qtd_produto_estoque > 0

Lista Usuarios Entregadores (Gerente X)
SELECT *
FROM usuarios INNER JOIN supervisores
WHERE usuarios.id_usuario = supervisores.get_id_usuario_entregador
AND supervisores.get_id_usuario_gerente = X

Vendas do dia
SELECT V.ordem_venda AS Venda,
SUM(V.valor_produto_venda_und * V.qtd_produto_venda) AS Valor
FROM vendas as V
WHERE V.data_venda = CURDATE()
GROUP BY V.ordem_venda

Balanço do dia
SELECT SUM(V.valor_produto_venda_und * V.qtd_produto_venda)
-
SUM(E.valor_produto_compra * E.qtd_produto_estoque) AS Valor
FROM vendas as V INNER JOIN produtos_estoque AS E
WHERE V.data_venda = CURDATE()
AND E.data_entrada_estoque = CURDATE()
